Example
const { fetchData, loading, error } = useFetch();

const getData = async () => {
  const token = await auth.currentUser.getIdToken();
  const response = await fetchData(
    'http://localhost:3000/api/v1/user/languages',
    'GET',
    null,
    token
  );
};
